#install.packages("timetk")
library(timetk)

seatbelts_df<-tk_tbl(Seatbelts) %>% 
  rownames_to_column("MonthsN") %>% 
  mutate(law=as.factor(law))

This gives us the number of traffic deaths over time in UK.

ggplot(seatbelts_df)+
